select * from production.brands b
join production.products p
on b.brand_id=p.brand_id
join production.stocks t
on t.product_id=p.product_id
join production.categories c
on c.category_id=p.category_id
join sales.order_items i
on i.item_id = p.product_id
join sales.orders o
on o.order_id = i.product_id
join sales.stores s
on s.store_id = o.store_id
join production.stocks
on t.store_id = s.store_id
join sales.staffs st
on st.staff_id = o.staff_id
join sales.customers cu
on cu.customer_id = o.customer_id
